Q. Корень зла
```python
a, b, c = [float(input()) for _ in range(3)]
if a == 0:
    if b == 0:
        if c == 0:
            print('Infinite solutions')
            quit()
        print('No solution')
        quit()
    print(f'{-c / b:.2f}')
else:
    d = b ** 2 - 4 * a * c
    if d < 0:
        print('No solution')
    elif d == 0:
        ans = -b / (2 * a)
        print(f'{ans:.2f}')
    else:
        x1 = (-b - d ** 0.5) / (2 * a)
        x2 = (-b + d ** 0.5) / (2 * a)
        print(f'{min(x1, x2):.2f} {max(x1, x2):.2f}')
```
